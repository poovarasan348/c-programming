#include <stdio.h>

int main() {
    char name[50], dept[30];
    int roll_no, semester;
    float marks[5], total = 0.0, average;
    char grade;
    printf("Enter your name: ");
    fgets(name, sizeof(name), stdin);
    printf("Enter your roll number: ");
    scanf("%d", &roll_no);
    printf("Enter your department: ");
    scanf(" %[^\n]s", dept);
    printf("Enter your semester: ");
    scanf("%d", &semester);
    printf("Enter marks for 5 subjects:\n");
    for (int i = 0; i < 5; i++) {
        printf("Subject %d marks: ", i + 1);
        scanf("%f", &marks[i]);
        total += marks[i];
    }
    average = total / 5;
    if (average >= 90) {
        grade = 'A';
    } else if (average >= 80) {
        grade = 'B';
    } else if (average >= 70) {
        grade = 'C';
    } else if (average >= 60) {
        grade = 'D';
    } else {
        grade = 'F';
    }
    printf("\n----- Student Details -----\n");
    printf("Name: %s", name);
    printf("Roll Number: %d\n", roll_no);
    printf("Department: %s\n", dept);
    printf("Semester: %d\n", semester);
    printf("Marks: ");
    for (int i = 0; i < 5; i++) {
        printf("%.2f ", marks[i]);
    }
    printf("\n");
    printf("Total Marks: %.2f\n", total);
    printf("Average Marks: %.2f\n", average);
    printf("Grade: %c\n", grade);
    return 0;
}
