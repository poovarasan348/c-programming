#include <stdio.h>
#include <string.h>
#include <ctype.h> 
int main() {
    char originalString[100]; 
    char lowerCaseString[100];
    char upperCaseString[100];
    int i;
    printf("Enter a string: ");
    fgets(originalString, sizeof(originalString), stdin);
    originalString[strcspn(originalString, "\n")] = 0;
    for (i = 0; originalString[i] != '\0'; i++) {
        lowerCaseString[i] = tolower(originalString[i]);
    }
    lowerCaseString[i] = '\0'; 
    for (i = 0; originalString[i] != '\0'; i++) {
        upperCaseString[i] = toupper(originalString[i]);
    }
    upperCaseString[i] = '\0';
    printf("Original String: %s\n", originalString);
    printf("Lowercase String: %s\n", lowerCaseString);
    printf("Uppercase String: %s\n", upperCaseString);
    return 0;
}
